2. Technologies

2.1. Fast multiplication

Fast multiplication uses bit manipulation to accelerate multiplication. 

#include<stdio.h>
int main () {
    int a = -35;
    int b = -2;
    int quotient = 0;
    while (a <= b) {
    int base = 1;
    int divisor = b;
    while (a - divisor <= divisor) {
        divisor <<= 1;
        base <<= 1;
    }
    quotient += base;
    a -= divisor;
    }
    printf("%d\n", quotient);
    return 0;
}

2.2. Fast exponentiation

Fast exponentiation uses multiplication of the same number to accelerate. 

#include <iostream>
using namespace std;
int main() {
  double x = 2.;
  int n = 7;
  bool isFraction = false;
  if (n < 0) {
    n = -n;
    isFraction = true;
  }
  int exp = 0;
  double res = 1.;
  while (exp < n) {
    int tmpExp = 1;
    double base = x;
    while (tmpExp < ((n-exp)>>1)) {
      base *= base;
      tmpExp <<= 1;
    }
    exp += tmpExp;
    res *= base;
  }
  cout << (isFraction ? 1.0/res : res) << endl;
  return 0;
}

2.3. Brain Kernighan’s Algorithm

When we subtract any number by 1, it inverts all the bits after the rightmost set bit(inclusive). 101(5) -> 100(4) 1000(8) -> 0111(7) Usage: x&(x-1) = invert the rightmost set bit

2.9. Trie

class Trie {
    private Trie[] children;
    private boolean isLeaf;
    public Trie() {
        this.children = new Trie[26];
        this.isLeaf = false;
    }
    public void insert(String word) {
        Trie node = this.root;
        for (char c : word.toCharArray()) {
            int index = c - 'a';
            if (node.children[index] == null) {
                node.children[index] = new Trie();
            }
            node = node.children[index];
        }
        node.isLeaf = true;
    }
    public boolean search(String word) {
        Trie node = this.root;
        for (char c : word.toCharArray()) {
            int index = c - 'a';
            if (node.children[index] == null) return false;
            node = node.children[index];
        }
        return node.isLeaf;
    }
}

2.11. Merge Sort

private void mergeSort(int[] nums, int left, int right, int[] tmp) {
    if (left >= right) return;
    int mid = left + ((right - left) >> 1);
    mergeSort(nums, left, mid, tmp);
    mergeSort(nums, mid+1, right, tmp);
    merge(nums, left, mid, right, tmp);
}
private void merge(int[] nums, int left, int mid, int right, int[] tmp) {
    int tmp1 = left, tmp2 = mid + 1;
    int i = left;
    while (tmp1 <= mid && tmp2 <= right) {
        if (nums[tmp1] < nums[tmp2]) {
            tmp[i++] = nums[tmp1++];
        }
        else {
            tmp[i++] = nums[tmp2++];
        }
    }
    while (tmp1 <= mid) {
        tmp[i++] = nums[tmp1++];
    }
    while (tmp2 <= right) {
        tmp[i++] = nums[tmp2++];
    }
    for (int k=left; k<=right; ++k) {
        nums[k] = tmp[k];
    }
}

2.12. Backtrack

public void combination(int[] nums) {
    List<List<Integer>> res = new ArrayList<>();
    LinkedList<Integer> comb = new LinkedList<>();
    backtrack(nums, res, comb, 0);
    return res;
}
private void backtrack(int[] nums, List<List<Integer>> res, LinkedList<Integer> comb, int start) {
    res.add(new LinkedList<Integer>(comb));
    for (int i=start; i<nums.length; ++i) {
        comb.add(nums[i]);
        backtrack(nums, res, comb, i+1);
        comb.removeLast();
    }
}
